@Override
protected CacheBuilder initCache(ConfigTree ct, String area) {
    String mode = ct.getProperty("mode", "standalone");

    RedisSetup setup = RedisSetupFactory.create(ct, mode);

    StatefulConnection<byte[], byte[]> connection = setup.createConnection(ct);
    StatefulRedisPubSubConnection<byte[], byte[]> pubSub =
            setup.isPubSubEnabled(ct) ? setup.createPubSubConnection() : null;

    ExternalCacheBuilder builder = buildCache(ct, setup.getClient(), connection, pubSub);

    register(area, setup.getClient(), connection, pubSub);

    return builder;
}

interface RedisSetup {
    AbstractRedisClient getClient();

    StatefulConnection<byte[], byte[]> createConnection(ConfigTree ct);

    StatefulRedisPubSubConnection<byte[], byte[]> createPubSubConnection();

    boolean isPubSubEnabled(ConfigTree ct);
}


class RedisSetupFactory {

    static RedisSetup create(ConfigTree ct, String mode) {
        switch (mode.toLowerCase()) {
            case "cluster":
                return new ClusterSetup(ct);
            case "sentinel":
                return new SentinelSetup(ct);
            default:
                return new StandaloneSetup(ct);
        }
    }
}

class StandaloneSetup implements RedisSetup {

    private final RedisClient client;
    private final RedisURI uri;

    StandaloneSetup(ConfigTree ct) {
        this.uri = buildUri(ct);
        this.client = RedisClient.create();

        client.setOptions(ClientOptions.builder()
                .disconnectedBehavior(ClientOptions.DisconnectedBehavior.REJECT_COMMANDS)
                .build());
    }

    @Override
    public AbstractRedisClient getClient() {
        return client;
    }

    @Override
    public StatefulConnection<byte[], byte[]> createConnection(ConfigTree ct) {
        return MasterReplica.connect(client, new JetCacheCodec(), Collections.singletonList(uri));
    }

    @Override
    public StatefulRedisPubSubConnection<byte[], byte[]> createPubSubConnection() {
        return client.connectPubSub(new JetCacheCodec(), uri);
    }

    @Override
    public boolean isPubSubEnabled(ConfigTree ct) {
        return parseBroadcastChannel(ct) != null;
    }

    private RedisURI buildUri(ConfigTree ct) {
        String host = ct.getProperty("host", "127.0.0.1");
        int port = Integer.parseInt(ct.getProperty("port", "6379"));
        int db = Integer.parseInt(ct.getProperty("database", "0"));
        String password = ct.getProperty("password");

        RedisURI.Builder builder = RedisURI.Builder.redis(host, port).withDatabase(db);
        if (password != null) {
            builder.withPassword(password.toCharArray());
        }
        return builder.build();
    }
}

class ClusterSetup implements RedisSetup {

    private final RedisClusterClient client;

    ClusterSetup(ConfigTree ct) {
        List<RedisURI> uris = buildUris(ct);
        this.client = RedisClusterClient.create(uris);
        client.setOptions(buildClusterOptions(ct));
    }

    @Override
    public AbstractRedisClient getClient() {
        return client;
    }

    @Override
    public StatefulConnection<byte[], byte[]> createConnection(ConfigTree ct) {
        return client.connect(new JetCacheCodec());
    }

    @Override
    public StatefulRedisPubSubConnection<byte[], byte[]> createPubSubConnection() {
        return client.connectPubSub(new JetCacheCodec());
    }

    @Override
    public boolean isPubSubEnabled(ConfigTree ct) {
        return parseBroadcastChannel(ct) != null;
    }

    private List<RedisURI> buildUris(ConfigTree ct) {
        String nodes = ct.getProperty("nodes");
        if (nodes == null) {
            throw new CacheConfigException("nodes required for cluster");
        }

        return Arrays.stream(nodes.split(","))
                .map(this::toUri)
                .collect(Collectors.toList());
    }

    private RedisURI toUri(String node) {
        String[] parts = node.split(":");
        return RedisURI.Builder.redis(parts[0], Integer.parseInt(parts[1])).build();
    }
}

private ExternalCacheBuilder buildCache(
        ConfigTree ct,
        AbstractRedisClient client,
        StatefulConnection<byte[], byte[]> connection,
        StatefulRedisPubSubConnection<byte[], byte[]> pubSub) {

    ExternalCacheBuilder builder = RedisLettuceCacheBuilder.createRedisLettuceCacheBuilder()
            .connection(connection)
            .pubSubConnection(pubSub)
            .redisClient(client)
            .asyncResultTimeoutInMillis(
                    ct.getProperty("asyncResultTimeoutInMillis",
                            CacheConsts.ASYNC_RESULT_TIMEOUT.toMillis())
            );

    parseGeneralConfig(builder, ct);
    return builder;
}

private void register(
        String area,
        AbstractRedisClient client,
        StatefulConnection<byte[], byte[]> connection,
        StatefulRedisPubSubConnection<byte[], byte[]> pubSub) {

    LettuceConnectionManager.defaultManager().init(client, connection);

    Map<String, Object> container = autoConfigureBeans.getCustomContainer();

    container.put(area + ".client", client);
    container.put(area + ".connection", connection);

    if (pubSub != null) {
        container.put(area + ".pubSubConnection", pubSub);
    }
}

class SentinelSetup implements RedisSetup {

    private final RedisClient client;
    private final RedisURI sentinelUri;

    SentinelSetup(ConfigTree ct) {
        this.sentinelUri = buildSentinelUri(ct);

        this.client = RedisClient.create();
        client.setOptions(ClientOptions.builder()
                .disconnectedBehavior(
                        ClientOptions.DisconnectedBehavior.REJECT_COMMANDS)
                .build());
    }

    @Override
    public AbstractRedisClient getClient() {
        return client;
    }

    @Override
    public StatefulConnection<byte[], byte[]> createConnection(ConfigTree ct) {
        StatefulRedisMasterReplicaConnection<byte[], byte[]> connection =
                MasterReplica.connect(
                        client,
                        new JetCacheCodec(),
                        Collections.singletonList(sentinelUri)
                );

        applyReadFrom(ct, connection);
        return connection;
    }

    @Override
    public StatefulRedisPubSubConnection<byte[], byte[]> createPubSubConnection() {
        return client.connectPubSub(new JetCacheCodec(), sentinelUri);
    }

    @Override
    public boolean isPubSubEnabled(ConfigTree ct) {
        return parseBroadcastChannel(ct) != null;
    }

    private RedisURI buildSentinelUri(ConfigTree ct) {
        String masterId = require(ct, "masterId");
        String nodes = require(ct, "nodes");

        int database = Integer.parseInt(ct.getProperty("database", "0"));
        String password = ct.getProperty("password");

        RedisURI.Builder builder = RedisURI.Builder
                .sentinel(masterId)
                .withDatabase(database);

        if (password != null && !password.isBlank()) {
            builder.withPassword(password.toCharArray());
        }

        for (String node : nodes.split(",")) {
            HostPort hp = parseHostPort(node.trim());
            builder.withSentinel(hp.host(), hp.port());
        }

        return builder.build();
    }

    private void applyReadFrom(
            ConfigTree ct,
            StatefulRedisMasterReplicaConnection<byte[], byte[]> connection) {

        String readFrom = ct.getProperty("readFrom");

        if (readFrom != null && !readFrom.isBlank()) {
            connection.setReadFrom(ReadFrom.valueOf(readFrom.trim()));
        }
    }

    private String require(ConfigTree ct, String key) {
        String value = ct.getProperty(key);

        if (value == null || value.isBlank()) {
            throw new CacheConfigException(
                    key + " property is required for sentinel mode");
        }

        return value;
    }

    private HostPort parseHostPort(String node) {
        String[] parts = node.split(":");

        if (parts.length != 2) {
            throw new CacheConfigException(
                    "Invalid sentinel node format: " + node);
        }

        return new HostPort(parts[0], Integer.parseInt(parts[1]));
    }

    private record HostPort(String host, int port) {
    }
}
