@Configuration
public class JetCacheRedisConfig {

    @Autowired
    private RedisProperties redisProperties;

    @Bean(name = "defaultClient")
    // Only loads if spring.redis.sentinel.master is defined in application.yml
    @ConditionalOnProperty(name = "spring.redis.sentinel.master")
    public RedisClient redisClient() {
        RedisProperties.Sentinel sentinelProps = redisProperties.getSentinel();
        
        // Custom logic to handle separate Sentinel passwords
        RedisURI.Builder builder = RedisURI.Builder
            .sentinel(sentinelProps.getNodes().get(0).split(":")[0], 
                      Integer.parseInt(sentinelProps.getNodes().get(0).split(":")[1]))
            .withSentinelMasterId(sentinelProps.getMaster())
            .withPassword(redisProperties.getPassword().toCharArray())
            .withDatabase(redisProperties.getDatabase());

        // Add other nodes and apply Sentinel-specific password
        sentinelProps.getNodes().stream().skip(1).forEach(node -> {
            String[] parts = node.split(":");
            builder.withSentinel(parts[0], Integer.parseInt(parts[1]));
        });

        RedisURI redisUri = builder.build();
        if (sentinelProps.getPassword() != null) {
            redisUri.getSentinels().forEach(s -> s.setPassword(sentinelProps.getPassword()));
        }

        return RedisClient.create(redisUri);
    }
}

@Configuration
public class JetCacheRedisConfig {

    private final RedisProperties redisProperties;

    public JetCacheRedisConfig(RedisProperties redisProperties) {
        this.redisProperties = redisProperties;
    }

    @Bean(name = "defaultClient")
    @ConditionalOnProperty(name = "spring.redis.sentinel.master")
    public RedisClient redisClient() {
        RedisProperties.Sentinel sentinel = redisProperties.getSentinel();
        
        // Initialize the builder with the first node and master ID
        RedisURI.Builder builder = createBaseBuilder(sentinel);

        // Apply global properties
        builder.withPassword(redisProperties.getPassword())
               .withDatabase(redisProperties.getDatabase());

        // Build and apply Sentinel-specific credentials if they exist
        RedisURI redisUri = builder.build();
        Optional.ofNullable(sentinel.getPassword())
                .ifPresent(pw -> redisUri.getSentinels().forEach(s -> s.setPassword(pw)));

        return RedisClient.create(redisUri);
    }

    private RedisURI.Builder createBaseBuilder(RedisProperties.Sentinel sentinel) {
        if (sentinel.getNodes() == null || sentinel.getNodes().isEmpty()) {
            throw new IllegalStateException("Sentinel nodes must be configured when master is defined.");
        }

        RedisURI.Builder builder = null;

        for (String node : sentinel.getNodes()) {
            String[] parts = node.split(":");
            String host = parts[0];
            int port = (parts.length > 1) ? Integer.parseInt(parts[1]) : 26379; // Default Sentinel port

            if (builder == null) {
                builder = RedisURI.Builder.sentinel(host, port, sentinel.getMaster());
            } else {
                builder.withSentinel(host, port);
            }
        }

        return builder;
    }
}

import io.lettuce.core.RedisClient;
import io.lettuce.core.RedisURI;
import org.springframework.boot.autoconfigure.data.redis.RedisProperties;
import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import java.util.Optional;

/**
 * Configuration class for JetCache Redis integration using Lettuce.
 * <p>
 * This configuration specifically handles Redis Sentinel deployments and allows for
 * the distinct configuration of master and sentinel credentials, which is common in
 * secured enterprise environments.
 */
@Configuration
public class JetCacheRedisConfig {

    private final RedisProperties redisProperties;

    /**
     * Constructs the configuration with Spring's injected Redis properties.
     *
     * @param redisProperties the auto-configured Redis properties from application.yml/properties
     */
    public JetCacheRedisConfig(RedisProperties redisProperties) {
        this.redisProperties = redisProperties;
    }

    /**
     * Creates a {@link RedisClient} bean configured for Sentinel mode.
     * <p>
     * This bean is only initialized if {@code spring.redis.sentinel.master} is present in
     * the environment configuration. It maps standard Spring Redis properties to the 
     * Lettuce-specific URI structure.
     *
     * @return a configured Lettuce {@link RedisClient}
     * @throws IllegalStateException if sentinel nodes are missing while sentinel mode is active
     */
    @Bean(name = "defaultClient")
    @ConditionalOnProperty(name = "spring.redis.sentinel.master")
    public RedisClient redisClient() {
        RedisProperties.Sentinel sentinel = redisProperties.getSentinel();
        
        // Initialize the builder with the first node and master ID
        RedisURI.Builder builder = createBaseBuilder(sentinel);

        // Apply global database and password settings
        builder.withPassword(redisProperties.getPassword())
               .withDatabase(redisProperties.getDatabase());

        // Build and apply Sentinel-specific credentials if they exist
        RedisURI redisUri = builder.build();
        Optional.ofNullable(sentinel.getPassword())
                .ifPresent(pw -> redisUri.getSentinels().forEach(s -> s.setPassword(pw)));

        return RedisClient.create(redisUri);
    }

    /**
     * Parses the configured sentinel nodes and initializes the URI builder.
     * <p>
     * It iterates through the list of "host:port" strings. If a port is omitted, 
     * it defaults to the standard Redis Sentinel port (26379).
     *
     * @param sentinel the sentinel configuration properties
     * @return a {@link RedisURI.Builder} populated with all sentinel node addresses
     * @throws IllegalStateException if the node list is null or empty
     */
    private RedisURI.Builder createBaseBuilder(RedisProperties.Sentinel sentinel) {
        if (sentinel.getNodes() == null || sentinel.getNodes().isEmpty()) {
            throw new IllegalStateException("Sentinel nodes must be configured when master is defined.");
        }

        RedisURI.Builder builder = null;

        for (String node : sentinel.getNodes()) {
            String[] parts = node.split(":");
            String host = parts[0];
            int port = (parts.length > 1) ? Integer.parseInt(parts[1]) : 26379;

            if (builder == null) {
                // The first node defines the starting point for the builder
                builder = RedisURI.Builder.sentinel(host, port, sentinel.getMaster());
            } else {
                // Subsequent nodes are added as additional sentinels
                builder.withSentinel(host, port);
            }
        }

        return builder;
    }
}

package com.example.config;

import com.alicp.jetcache.CacheBuilder;
import com.alicp.jetcache.redis.lettuce.JetCacheCodec;
import com.alicp.jetcache.redis.lettuce.RedisLettuceCacheBuilder;
import com.alicp.jetcache.support.Fastjson2KeyConvertor;
import com.alicp.jetcache.support.FastjsonKeyConvertor;
import com.alicp.jetcache.support.JacksonKeyConvertor;
import com.alicp.jetcache.support.KeyConvertor;
import io.lettuce.core.AbstractRedisClient;
import io.lettuce.core.RedisClient;
import io.lettuce.core.RedisURI;
import io.lettuce.core.api.StatefulConnection;
import io.lettuce.core.masterreplica.MasterReplica;
import io.lettuce.core.pubsub.StatefulRedisPubSubConnection;
import org.springframework.boot.autoconfigure.condition.ConditionalOnBean;
import org.springframework.boot.autoconfigure.data.redis.RedisProperties;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.core.env.Environment;

/**
 * Custom Redis configuration for JetCache using Lettuce.
 *
 * <p>
 * This configuration replaces JetCache's default Redis auto-configuration to support:
 * <ul>
 *     <li>Redis Sentinel with authentication (Redis + Sentinel passwords)</li>
 *     <li>Standalone Redis (for local development)</li>
 *     <li>JetCache broadcastChannel (Pub/Sub)</li>
 *     <li>Configurable keyConvertor via application.yml</li>
 * </ul>
 *
 * <p>
 * Why this exists:
 * <ul>
 *     <li>JetCache 2.7.x does NOT support Sentinel password in its auto-config</li>
 *     <li>We rebuild RedisURI from Spring configuration to ensure full compatibility</li>
 * </ul>
 *
 * <p>
 * Behaviour:
 * <ul>
 *     <li>If {@code spring.redis.sentinel.nodes} exists → Sentinel mode</li>
 *     <li>Otherwise → standalone mode</li>
 * </ul>
 *
 * <p>
 * IMPORTANT:
 * <ul>
 *     <li>You MUST exclude RedisLettuceAutoConfiguration</li>
 *     <li>This class becomes the single source of truth for JetCache Redis setup</li>
 * </ul>
 */
@Configuration
@ConditionalOnBean(RedisProperties.class)
public class JetCacheRedisConfig {

    /**
     * Build RedisURI from Spring RedisProperties.
     *
     * <p>
     * Supports both Sentinel and standalone modes.
     */
    @Bean
    public RedisURI jetCacheRedisURI(RedisProperties redisProperties) {

        RedisProperties.Sentinel sentinel = redisProperties.getSentinel();

        // Sentinel mode
        if (sentinel != null && sentinel.getNodes() != null && !sentinel.getNodes().isEmpty()) {

            RedisURI.Builder builder = RedisURI.builder()
                    .withSentinelMasterId(sentinel.getMaster());

            for (String node : sentinel.getNodes()) {
                String[] parts = node.split(":");
                builder.withSentinel(parts[0], Integer.parseInt(parts[1]));
            }

            if (redisProperties.getPassword() != null) {
                builder.withPassword(redisProperties.getPassword().toCharArray());
            }

            if (sentinel.getPassword() != null) {
                builder.withSentinelPassword(sentinel.getPassword().toCharArray());
            }

            builder.withDatabase(redisProperties.getDatabase());
            return builder.build();
        }

        // Standalone mode
        RedisURI.Builder builder = RedisURI.builder()
                .withHost(redisProperties.getHost() != null ? redisProperties.getHost() : "localhost")
                .withPort(redisProperties.getPort() != null ? redisProperties.getPort() : 6379)
                .withDatabase(redisProperties.getDatabase());

        if (redisProperties.getPassword() != null) {
            builder.withPassword(redisProperties.getPassword().toCharArray());
        }

        return builder.build();
    }

    /**
     * Create Redis client.
     *
     * <p>
     * Sentinel mode uses dynamic topology resolution.
     * Standalone mode binds directly to the URI.
     */
    @Bean(destroyMethod = "shutdown")
    public AbstractRedisClient jetCacheRedisClient(RedisURI redisURI) {
        if (redisURI.getSentinelMasterId() != null) {
            return RedisClient.create();
        } else {
            return RedisClient.create(redisURI);
        }
    }

    /**
     * Main Redis connection for cache operations.
     *
     * <p>
     * Sentinel mode → MasterReplica (auto failover)
     * Standalone mode → direct connection
     */
    @Bean(destroyMethod = "close")
    public StatefulConnection<byte[], byte[]> jetCacheConnection(
            AbstractRedisClient client,
            RedisURI redisURI) {

        if (redisURI.getSentinelMasterId() != null) {
            return MasterReplica.connect(client, new JetCacheCodec(), redisURI);
        } else {
            return ((RedisClient) client).connect(new JetCacheCodec(), redisURI);
        }
    }

    /**
     * Pub/Sub connection for JetCache broadcastChannel.
     *
     * <p>
     * Required for distributed cache invalidation.
     * Uses the same RedisURI to ensure identical auth + topology.
     */
    @Bean(destroyMethod = "close")
    public StatefulRedisPubSubConnection<byte[], byte[]> jetCachePubSubConnection(
            AbstractRedisClient client,
            RedisURI redisURI) {

        return ((RedisClient) client).connectPubSub(new JetCacheCodec(), redisURI);
    }

    /**
     * Resolve KeyConvertor from configuration.
     *
     * <p>
     * Reads:
     * {@code jetcache.remote.default.keyConvertor}
     *
     * <p>
     * Supported values:
     * <ul>
     *     <li>fastjson2 (default)</li>
     *     <li>fastjson</li>
     *     <li>jackson</li>
     *     <li>none</li>
     *     <li>fully qualified class name</li>
     * </ul>
     */
    private KeyConvertor resolveKeyConvertor(Environment env) {
        String key = env.getProperty("jetcache.remote.default.keyConvertor", "fastjson2");

        switch (key.toLowerCase()) {
            case "fastjson2":
                return new Fastjson2KeyConvertor();
            case "fastjson":
                return new FastjsonKeyConvertor();
            case "jackson":
                return new JacksonKeyConvertor();
            case "none":
                return KeyConvertor.NONE;
            default:
                try {
                    return (KeyConvertor) Class.forName(key)
                            .getDeclaredConstructor()
                            .newInstance();
                } catch (Exception e) {
                    throw new IllegalArgumentException("Invalid keyConvertor: " + key, e);
                }
        }
    }

    /**
     * JetCache Redis builder.
     *
     * <p>
     * Wires:
     * <ul>
     *     <li>data connection</li>
     *     <li>pub/sub connection</li>
     *     <li>Redis client</li>
     *     <li>keyConvertor from config</li>
     * </ul>
     */
    @Bean
    public CacheBuilder<?> jetCacheRemoteCacheBuilder(
            StatefulConnection<byte[], byte[]> connection,
            StatefulRedisPubSubConnection<byte[], byte[]> pubSubConnection,
            AbstractRedisClient client,
            Environment env) {

        return RedisLettuceCacheBuilder.createRedisLettuceCacheBuilder()
                .connection(connection)
                .pubSubConnection(pubSubConnection)
                .redisClient(client)
                .keyConvertor(resolveKeyConvertor(env))
                .valueEncoder(JetCacheCodec.INSTANCE::encode)
                .valueDecoder(JetCacheCodec.INSTANCE::decode);
    }
}

package com.example.config;

import com.alicp.jetcache.CacheBuilder;
import com.alicp.jetcache.redis.lettuce.JetCacheCodec;
import com.alicp.jetcache.redis.lettuce.RedisLettuceCacheBuilder;
import com.alicp.jetcache.support.FastjsonKeyConvertor;
import com.alicp.jetcache.support.JacksonKeyConvertor;
import com.alicp.jetcache.support.JavaValueDecoder;
import com.alicp.jetcache.support.JavaValueEncoder;
import com.alicp.jetcache.support.KeyConvertor;
import io.lettuce.core.AbstractRedisClient;
import io.lettuce.core.RedisClient;
import io.lettuce.core.RedisURI;
import io.lettuce.core.api.StatefulConnection;
import io.lettuce.core.masterreplica.MasterReplica;
import io.lettuce.core.pubsub.StatefulRedisPubSubConnection;
import org.springframework.boot.autoconfigure.condition.ConditionalOnBean;
import org.springframework.boot.autoconfigure.data.redis.RedisProperties;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.core.env.Environment;

/**
 * Custom Redis configuration for JetCache using Lettuce.
 * * <p>Why this exists:
 * JetCache 2.7.9's {@code RedisLettuceAutoConfiguration} instantiates its own {@link RedisClient}
 * and does not support separate passwords for Sentinel nodes. This configuration manually
 * wires the JetCache beans to support complex authentication.
 * * <p>Features:
 * <ul>
 * <li>Full support for Sentinel node authentication (Sentinel Password).</li>
 * <li>Standard Redis password support for Data nodes.</li>
 * <li>Dynamic switching between Sentinel and Standalone based on {@code RedisProperties}.</li>
 * <li>Manual injection of {@link RedisLettuceCacheBuilder} to bypass auto-init issues.</li>
 * </ul>
 */
@Configuration
@ConditionalOnBean(RedisProperties.class)
public class JetCacheRedisConfig {

    /**
     * Builds the {@link RedisURI} from Spring's {@link RedisProperties}.
     * * <p>In Sentinel mode, it iterates through the sentinel nodes and manually sets the
     * password for each node, as the Lettuce Builder does not provide a global 
     * {@code withSentinelPassword} method.
     * * @param redisProperties Injected Spring Boot Redis properties
     * @return Fully configured RedisURI with necessary credentials
     */
    @Bean
    public RedisURI jetCacheRedisURI(RedisProperties redisProperties) {
        RedisProperties.Sentinel sentinel = redisProperties.getSentinel();

        // Check if Sentinel is configured
        if (sentinel != null && sentinel.getNodes() != null && !sentinel.getNodes().isEmpty()) {
            RedisURI.Builder builder = RedisURI.builder()
                    .withSentinelMasterId(sentinel.getMaster());

            for (String node : sentinel.getNodes()) {
                String[] parts = node.split(":");
                String host = parts[0];
                int port = (parts.length > 1) ? Integer.parseInt(parts[1]) : 26379;
                builder.withSentinel(host, port);
            }

            // Set Data Node Password
            if (redisProperties.getPassword() != null) {
                builder.withPassword(redisProperties.getPassword().toCharArray());
            }

            RedisURI finalUri = builder.build();

            // Manually inject Sentinel Password into each node's URI
            if (sentinel.getPassword() != null) {
                finalUri.getSentinels().forEach(s -> s.setPassword(sentinel.getPassword()));
            }

            finalUri.setDatabase(redisProperties.getDatabase());
            return finalUri;
        }

        // Fallback: Standalone Mode
        RedisURI.Builder builder = RedisURI.builder()
                .withHost(redisProperties.getHost() != null ? redisProperties.getHost() : "localhost")
                .withPort(redisProperties.getPort() != null ? redisProperties.getPort() : 6379)
                .withDatabase(redisProperties.getDatabase());

        if (redisProperties.getPassword() != null) {
            builder.withPassword(redisProperties.getPassword().toCharArray());
        }

        return builder.build();
    }

    /**
     * Creates the Lettuce RedisClient.
     * * @param redisURI The authenticated URI
     * @return A thread-safe RedisClient
     */
    @Bean(destroyMethod = "shutdown")
    public AbstractRedisClient jetCacheRedisClient(RedisURI redisURI) {
        return RedisClient.create(redisURI);
    }

    /**
     * Creates the primary connection for cache data operations.
     * * <p>If Sentinel is detected, it uses {@link MasterReplica#connect} to ensure
     * automatic failover and topology discovery.
     * * @param client The Redis client
     * @param redisURI The configured URI
     * @return Connection for Redis commands
     */
    @Bean(destroyMethod = "close")
    public StatefulConnection<byte[], byte[]> jetCacheConnection(
            AbstractRedisClient client,
            RedisURI redisURI) {

        if (redisURI.getSentinelMasterId() != null) {
            // MasterReplica handles sentinel topology updates automatically
            return MasterReplica.connect((RedisClient) client, new JetCacheCodec(), redisURI);
        } else {
            return ((RedisClient) client).connect(new JetCacheCodec(), redisURI);
        }
    }

    /**
     * Creates a dedicated connection for Redis Pub/Sub.
     * Required for JetCache's remote cache invalidation (broadcastChannel).
     * * @param client The Redis client
     * @param redisURI The configured URI
     * @return Pub/Sub connection
     */
    @Bean(destroyMethod = "close")
    public StatefulRedisPubSubConnection<byte[], byte[]> jetCachePubSubConnection(
            AbstractRedisClient client,
            RedisURI redisURI) {
        return ((RedisClient) client).connectPubSub(new JetCacheCodec(), redisURI);
    }

    /**
     * Configures the JetCache CacheBuilder for Redis.
     * * <p>By providing this bean, JetCache's {@code RedisLettuceAutoInit} will recognize 
     * the manual configuration and use our custom connections.
     * * @param connection Primary data connection
     * @param pubSubConnection Secondary pub/sub connection
     * @param client The Lettuce client
     * @param env Spring Environment for property resolution
     * @return Configured RedisLettuceCacheBuilder
     */
    @Bean
    public CacheBuilder<?> jetCacheRemoteCacheBuilder(
            StatefulConnection<byte[], byte[]> connection,
            StatefulRedisPubSubConnection<byte[], byte[]> pubSubConnection,
            AbstractRedisClient client,
            Environment env) {

        return RedisLettuceCacheBuilder.createRedisLettuceCacheBuilder()
                .connection(connection)
                .pubSubConnection(pubSubConnection)
                .redisClient(client)
                .keyConvertor(resolveKeyConvertor(env))
                .valueEncoder(JavaValueEncoder.INSTANCE)
                .valueDecoder(JavaValueDecoder.INSTANCE);
    }

    /**
     * Resolves the {@link KeyConvertor} based on the property
     * {@code jetcache.remote.default.keyConvertor}.
     */
    private KeyConvertor resolveKeyConvertor(Environment env) {
        String key = env.getProperty("jetcache.remote.default.keyConvertor", "fastjson");

        return switch (key.toLowerCase()) {
            case "fastjson" -> FastjsonKeyConvertor.INSTANCE;
            case "jackson" -> JacksonKeyConvertor.INSTANCE;
            case "none" -> KeyConvertor.NONE;
            default -> {
                try {
                    yield (KeyConvertor) Class.forName(key).getDeclaredConstructor().newInstance();
                } catch (Exception e) {
                    yield FastjsonKeyConvertor.INSTANCE;
                }
            }
        };
    }
}

/**
     * Resolves the key converter function.
     * JetCache built-in converters (Fastjson, Jackson) implement Function<Object, Object>.
     */
    @SuppressWarnings("unchecked")
    private Function<Object, Object> resolveKeyConvertor(Environment env) {
        String key = env.getProperty("jetcache.remote.default.keyConvertor", "fastjson");

        return switch (key.toLowerCase()) {
            case "fastjson" -> FastjsonKeyConvertor.INSTANCE;
            case "jackson" -> JacksonKeyConvertor.INSTANCE;
            case "none" -> null; // JetCache handles null as no conversion
            default -> {
                try {
                    // Try to load custom Function class if provided
                    Class<?> clazz = Class.forName(key);
                    yield (Function<Object, Object>) clazz.getDeclaredConstructor().newInstance();
                } catch (Exception e) {
                    yield FastjsonKeyConvertor.INSTANCE;
                }
            }
        };


/**
 * Resolves the key converter function.
 * This version uses traditional switch syntax to ensure compatibility with Java 8/11.
 * * @param env Spring Environment
 * @return The resolved key conversion function
 */
@SuppressWarnings("unchecked")
private Function<Object, Object> resolveKeyConvertor(Environment env) {
    String key = env.getProperty("jetcache.remote.default.keyConvertor", "fastjson");

    switch (key.toLowerCase()) {
        case "fastjson":
            return FastjsonKeyConvertor.INSTANCE;
        case "jackson":
            return JacksonKeyConvertor.INSTANCE;
        case "none":
            return null; 
        default:
            try {
                // Try to load custom Function class if provided
                Class<?> clazz = Class.forName(key);
                return (Function<Object, Object>) clazz.getDeclaredConstructor().newInstance();
            } catch (Exception e) {
                // Log the error if necessary and fallback to default
                return FastjsonKeyConvertor.INSTANCE;
            }
    }
}
