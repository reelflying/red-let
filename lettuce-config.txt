@Configuration
public class JetCacheOverrideConfig {

    @Bean(name = RedisLettuceAutoConfiguration.AUTO_INIT_BEAN_NAME)
    public RedisLettuceAutoConfiguration.RedisLettuceAutoInit redisLettuceAutoInit() {
        return new MyRedisLettuceAutoInit();
    }
}


public class MyRedisLettuceAutoInit
        extends RedisLettuceAutoConfiguration.RedisLettuceAutoInit {

    @Autowired
    private RedisProps props;

    @Override
    protected CacheBuilder initCache(ConfigTree ct, String cacheAreaWithPrefix) {

        AbstractRedisClient client;
        StatefulConnection<byte[], byte[]> connection;

        if ("sentinel".equalsIgnoreCase(props.getMode())) {

            RedisURI.Builder builder = RedisURI.builder()
                    .withSentinelMasterId(props.getSentinelMaster());

            props.getSentinels().forEach(s -> {
                String[] hp = s.split(":");
                builder.withSentinel(hp[0], Integer.parseInt(hp[1]));
            });

            if (props.getPassword() != null) {
                builder.withPassword(props.getPassword().toCharArray());
            }

            RedisClient redisClient = RedisClient.create();

            client = redisClient;

            connection = MasterReplica.connect(
                    redisClient,
                    new JetCacheCodec(),
                    List.of(builder.build())
            );

        } else {
            // standalone
            RedisURI uri = RedisURI.builder()
                    .withHost(props.getHost())
                    .withPort(props.getPort())
                    .withPassword(props.getPassword() == null
                            ? null
                            : props.getPassword().toCharArray())
                    .build();

            RedisClient redisClient = RedisClient.create(uri);

            client = redisClient;
            connection = redisClient.connect(new JetCacheCodec());
        }

        ExternalCacheBuilder builder = RedisLettuceCacheBuilder
                .createRedisLettuceCacheBuilder()
                .connection(connection)
                .redisClient(client);

        parseGeneralConfig(builder, ct);

        return builder;
    }
}
