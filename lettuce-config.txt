@Configuration
public class JetCacheManualConfig {

    // Use the exact same bean name from the original class
    @Bean(name = RedisLettuceAutoConfiguration.AUTO_INIT_BEAN_NAME)
    public RedisLettuceAutoConfiguration.RedisLettuceAutoInit redisLettuceAutoInit() {
        return new RedisLettuceAutoConfiguration.RedisLettuceAutoInit() {
            
            @Override
            protected CacheBuilder initCache(ConfigTree ct, String cacheAreaWithPrefix) {
                // JETCACHE ORIGINAL CODE LOOKS FOR "uri" HERE.
                // WE BYPASS THAT BY BUILDING OUR OWN RedisURI.
                
                String host = ct.getProperty("host", "127.0.0.1");
                int port = Integer.parseInt(ct.getProperty("port", "6379"));
                String password = ct.getProperty("password");
                
                RedisURI.Builder builder = RedisURI.Builder.redis(host, port);
                if (password != null) {
                    builder.withPassword(password.toCharArray());
                }
                RedisURI redisUri = builder.build();

                // Build the client manually
                RedisClient client = RedisClient.create(redisUri);
                StatefulConnection<byte[], byte[]> connection = client.connect(new JetCacheCodec());

                // Build the JetCache-specific builder
                ExternalCacheBuilder externalCacheBuilder = RedisLettuceCacheBuilder
                        .createRedisLettuceCacheBuilder()
                        .connection(connection)
                        .redisClient(client);

                // This helper method is in the parent class to handle common jetcache props
                parseGeneralConfig(externalCacheBuilder, ct);

                return externalCacheBuilder;
            }
        };
    }
}
